22      @Test
23      public void testEncryptAndDecrypt() throws UnsupportedEncodingException {
24          PPK ppk = PPK.publicKey("/public.der").privateKey("/private.der").build();
25          byte[] encrypted = ppk.encrypt(content, StandardCharsets.UTF_8);
26          String decrypted = ppk.decrypt(encrypted, StandardCharsets.UTF_8);
27          assertEquals(content, decrypted);
28      }
